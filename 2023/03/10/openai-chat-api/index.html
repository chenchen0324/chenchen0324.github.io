import os
import openai
import sseclient

class ChatApp:
    def __init__(self):
        # Setting the API key to use the OpenAI API
        openai.api_key = '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'
        self.messages = [
            {"role": "system", "content": "You are a helpful assistant ."},
        ]

    def chat(self, user, message):
        self.messages.append({"role": "user", "content": message})
        response = openai.ChatCompletion.create(
            model="gpt-3.5-turbo",
            user = user,
            stream = True,
            messages=self.messages
        )
        
        ans = ''
        for event in response:
            if 'content' in event['choices'][0]['delta']:
                ans += event['choices'][0]['delta']['content']
                print(event['choices'][0]['delta']['content'], end="", flush=True)
        
        # enable here if you needs consecutive conversation
        self.messages.append({"role": "assistant", "content": ans})

print('User Name:')
user = input()
userPrompt = ChatApp()
while 1:
    print('\nPrompt:')
    prompt = input()
    print('\n')
    
    if prompt == 'end':
        break
    print('Assistant:')
    completion = userPrompt.chat(user, prompt)
    
    print('\n')
