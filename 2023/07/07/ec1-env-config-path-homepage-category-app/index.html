[EC1] Env + Config Path + Homepage + Category App

Env Prepare →

1, Install atom, Git

2, Install virtual env

python3 -m pip install --user virtualenv
# create a new
python3 -m venv {{ecom}}
# enter into virtual env
source {{ecom}}/bin/activate
# check all packages
pip freeze
# deactiavte virtual env
deactivate

3, install Django

pip install django==3.1

4, start project

djangp-admin startproject {{ec}}
cd {{ec}}
python manage.py runserver

Config Path + Homepage →

1, add url for homepage

# ec/ec/urls.py
path('', views.home, name='home'),

2, add views.py into ec/ec

3, import views

# ec/ec/urls.py
from . import views

4, test http in the views.py

# ec/ec/views.py
from django.http import HttpResponse
def home(request):
    return HttpResponse('Homepage')

5, create templates folder in root and create a home.html as the template

# ec/templates/home.html
<h2>This is template!!</h2>

6, Configure the template path in the settings.py

# ec/ec/settings.py
TEMPLATES = [
    {
        'BACKEND': 'django.template.backends.django.DjangoTemplates',
        'DIRS': ['templates'],
        'APP_DIRS': True,
        'OPTIONS': {
            'context_processors': [
                'django.template.context_processors.debug',
                'django.template.context_processors.request',
                'django.contrib.auth.context_processors.auth',
                'django.contrib.messages.context_processors.messages',
            ],
        },
    },
]

7, update ec/ec/views.py to use the template

# ec/ec/views.py
from django.shortcuts import render
def home(request):
    return render(request, 'home.html')

python manage.py startapp category

#ec/ec/settings.py
INSTALLED_APPS = [
    'django.contrib.admin',
    'django.contrib.auth',

    'django.contrib.staticfiles',
    'category',
]

3, Create model (database table) for category

#ec/category/models.py
from django.db import models

# Create your models here.
class Category(models.Model):
    category_name = models.CharField(max_length=50, unique=True)
    slug = models.SlugField(max_length=100, unique=True)
    description = models.TextField(max_length=255, blank=True)
    cat_image = models.ImageField(upload_to='photos/categories', blank=True)
    class Meta:
        verbose_name = 'category'
        verbose_name_plural = 'categories'
    def __str__(self):
        return self.category_name

4, add Category into admin

#ec/category/admin.py
from django.contrib import admin
from .models import Category

# Register your models here.

admin.site.register(Category)

5, Update the database

python manage.py makemigrations
python manage.py migrate

6, create admin for application

python manage.py createsuperuser

7, check Category table in http://127.0.0.1:8000/admin/
